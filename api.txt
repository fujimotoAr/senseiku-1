1. signup
    path: /signupTutor /signupStudent
    method: POST
    request: {
        "username": string, 
        "password": string,
        "email": string,
        "first_name": string
    }
    response: {
        "username": string, 
        "message": string
    }
2. login
    path: /loginTutor /loginStudent
    method: POST
    request: {
        "username": string, 
        "password": string
    }
    response: {
        "isAuth": boolean, 
        "username": string,
        "token": string,
        "message": string,
        "currentTime": string
    }
3. profile
    path: /profileTutor/{username} /profileStudent/{username}
    method: GET
    response: {
        "username": string, 
        "email": string,
        "first_name": string
    }
4. getNewCourse -> home
    path: /newCourse
    method: GET
    response: [
        {
            "id": integer,
            "course_name": string, 
            "description": string,
            "pricing": integer,
            "first_name": string
        },
        {
            "id": integer,
            "course_name": string, 
            "description": string,
            "pricing": integer,
            "first_name": string
        }
    ]
5. getMyCourse -> dashboard sensei

    method: GET
    response: [
        {

            "course_name": string, 
            "description": string,
            "pricing": integer,

        },
        {
            "id": integer,
            "course_name": string, 
            "description": string,
            "pricing": integer,
            "tutor_username": string
        }
    ]
6. addCourse -> dashboard sensei
    path: /addCourse
    method: POST
    request: {
        "course_name": string, 
        "description": string,
        "pricing": integer,
        "tutor_username": string
    }
    response: {
        "course_name": string, 
        "description": string,
        "pricing": integer,
        "tutor_username": string,
        "message": string
    }
7. deleteCourse -> dashboard sensei
    path: /deleteCourse/{id}
    method: DELETE
    response : {
        "id": integer,
        "message": string
    }
8. getCourseDetail
    path: /courseDetail/{id}
    method: GET
    response: {
        "id": integer,
        "course_name": string, 
        "description": string,
        "pricing": integer,
        "first_name": string,
        "day": string,
        "hour_start": string, (format: "HH:MM")
        "hour_finish": string
    }
9. tracker
    path: /tracker
    method: POST
    request: {
        "course_id": integer,
        "username": string,
        "event": integer,
        "timestamp": string
    }
    response: {
        "message": string
    }
